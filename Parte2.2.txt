void __interrupt() isr(void)
{    
    if (PIR1bits.RCIF)
    {
        //PORTA = RCREG;
        switch (RCREG)
        {
            case 49:
                PALABRA();
                if (PIR1bits.TXIF)
                {
                    TXREG = 13;
                }
                MENU();
                break;

            case 50:
                INGRESE_NUEVO_VALOR();
                do 
                {
                    if (PIR1bits.RCIF)
                    {
                        PORTA = RCREG;
                        Boolean = 255;
                    }
                }
                while (Boolean==0);
                __delay_ms(50);
                if (PIR1bits.TXIF)
                {
                    TXREG = 13;
                }
                __delay_ms(50);
                if (PIR1bits.TXIF)
                {
                    TXREG = 13;
                }
                MENU();
                Boolean=0;
                break;
            
            case 51:
                INGRESE_NUEVO_VALOR();
                do 
                {
                    if (PIR1bits.RCIF)
                    {
                        PORTB = RCREG;
                        Boolean = 255;
                    }
                }
                while (Boolean==0);
                __delay_ms(50);
                if (PIR1bits.TXIF)
                {
                    TXREG = 13;
                }
                __delay_ms(50);
                if (PIR1bits.TXIF)
                {
                    TXREG = 13;
                }
                MENU();
                Boolean=0;
                break;
        }
    }
}
